def display_random_images(x, y):
    num_classes = len(np.unique(y)) # Determine the number of classes
    
    fig, ax = plt.subplots(1, num_classes, figsize=(20, 5)) # Create subplots for each class

    for i in range(num_classes):
        class_indices = np.where(y == i)[0] # Get the indices of the current class
        random_index = np.random.choice(class_indices) # Choose a random index from the current class
        image = x[random_index] # Get the corresponding image
        ax[i].imshow(image, cmap='gray') # Display the image
        ax[i].set_title(f"Class {i}") # Set the title of the subplot to the class label
        ax[i].axis('off') # Hide the axis

    plt.show() # Show the plot

def exp_decay(epoch):
    initial_lr = 0.01
    k = 0.1
    lr = initial_lr * math.exp(-k * epoch)
    return lr

def build_model(hp):
    inputs = Input(shape=(28, 28))
    x = Flatten()(inputs)
    
    # Tune the number of units in the first dense layer
    hp_units_1 = hp.Int('units_1', min_value=256, max_value=512, step=32)
    x = Dense(units=hp_units_1, activation='relu')(x)
    
    # Tune the number of units in the second dense layer
    hp_units_2 = hp.Int('units_2', min_value=128, max_value=256, step=32)
    x = Dense(units=hp_units_2, activation='relu')(x)
    
    outputs = Dense(10, activation='softmax')(x)
    
    # Tune the learning rate for the optimizer
    hp_learning_rate = hp.Choice('learning_rate', values=[1e-2, 1e-3, 1e-4])
    
    model = Model(inputs=inputs, outputs=outputs)
    
    model.compile(optimizer=tf.keras.optimizers.Adam(learning_rate=hp_learning_rate),
                  loss='categorical_crossentropy',
                  metrics=['accuracy'])
    
    return model

# Define the Keras Tuner
tuner = RandomSearch(
    build_model,
    objective='val_accuracy',
    max_trials=10,
    executions_per_trial=3,
    directory='my_dir',
    project_name='mnist_tuner')

# Start the tuning process
tuner.search(x_train, y_train,
             validation_split=0.2,
             epochs=10,
